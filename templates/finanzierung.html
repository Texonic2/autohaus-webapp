def berechne_rate(preis, monate, anzahlung):
    darlehen = preis - anzahlung
    zinsrate = 0.05 # Beispiel: 5% Zinssatz
    raten = darlehen * (zinsrate * (1 + zinsrate) ** monate) / ((1 + zinsrate) ** monate - 1)
    return raten
